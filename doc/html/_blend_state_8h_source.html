363 
366  RenderTargetBlendDesc RenderTargets[MaxRenderTargets];

377  BlendStateDesc() :
378  AlphaToCoverageEnable(False),
379  IndependentBlendEnable(False)
380  {
381  }
382 
384 
393  bool operator == (const BlendStateDesc& RHS)const
394  {
395  bool bRTsEqual = true;
396  for( int i = 0; i < MaxRenderTargets; ++i )
397  if( !(RenderTargets[i] == RHS.RenderTargets[i]) )
398  {
399  bRTsEqual = false;
400  break;
401  }
402 
403  return bRTsEqual &&
404  AlphaToCoverageEnable == RHS.AlphaToCoverageEnable &&
405  IndependentBlendEnable== RHS.IndependentBlendEnable;
406  }
